26. Move Zeroes
Move all zeros to the end while keeping non-zero elements in original relative order. Must do this IN-PLACE.
💡 Key: Two-pointer approach where j tracks position for next non-zero element
Approach: Two Pointers
  • j tracks position for next non-zero
  • For each non-zero at i, swap with j position
  • Only increment j when placing non-zero
Solution
def move_zeroes(nums):
    j = 0
    for i in range(len(nums)):
        if nums[i] != 0:
            nums[j], nums[i] = nums[i], nums[j]
            j += 1

# Example: [0, 1, 0, 3, 12] → [1, 3, 12, 0, 0]
⏱️ Time
O(n)
💾 Space
O(1)

27. Can Jump (Jump Game)
From each position, you can jump UP TO nums[i] steps forward. Determine if you can reach the last index starting from index 0.
💡 Key: Greedy approach - track farthest reachable position at each step
Solution
def can_jump(nums):
    reach = 0
    for i, num in enumerate(nums):
        if i > reach:
            return False
        reach = max(reach, i + num)
    return True

# Example: [2, 3, 1, 1, 4] → True
⏱️ Time
O(n)
💾 Space
O(1)

28. Jump Game II (Minimum Jumps)
Find MINIMUM number of jumps to reach last index. Guaranteed you can always reach the end.
💡 Key: Track 'end' of current jump range and 'farthest' reachable position
Solution
def jump_game_ii(nums):
    jumps = end = farthest = 0
    for i in range(len(nums) - 1):
        farthest = max(farthest, i + nums[i])
        if i == end:
            jumps += 1
            end = farthest
    return jumps

# Example: [2, 3, 1, 1, 4] → 2 jumps
⏱️ Time
O(n)

29. Pivot Index (Balance Point)
Find the pivot index where sum of left elements equals sum of right elements.
💡 Key: Formula: left_sum == total - left_sum - nums[i]
Solution
def pivot_index(nums):
    total = sum(nums)
    left = 0
    for i, x in enumerate(nums):
        if left == total - left - x:
            return i
        left += x
    return -1

# Example: [1, 7, 3, 6, 5, 6] → 3
⏱️ Time
O(n)
💾 Space
O(1)

30. Maximum Length Subarray with Positive Product
Find max length subarray with positive product. Negative×Negative=Positive, Zero resets.
💡 Key: Track two states: pos (positive) and neg (negative) lengths
Solution
def max_len_positive_product(nums):
    pos = neg = ans = 0
    for num in nums:
        if num == 0:
            pos = neg = 0
        elif num > 0:
            pos += 1
            neg = neg + 1 if neg else 0
        else:
            new_pos = neg + 1 if neg else 0
            new_neg = pos + 1
            pos, neg = new_pos, new_neg
        ans = max(ans, pos)
    return ans
⏱️ Time
O(n)
💾 Space
O(1)

31. Remove Duplicates from Sorted Array II
Remove duplicates in-place, allowing each element to appear at most TWICE. Return new length.
💡 Key: Compare current element with element 2 positions back
Solution
def remove_duplicates_ii(nums):
    if len(nums) <= 2:
        return len(nums)
    idx = 2
    for i in range(2, len(nums)):
        if nums[i] != nums[idx - 2]:
            nums[idx] = nums[i]
            idx += 1
    return idx

33. Maximum Average Subarray I
Find maximum average of any contiguous subarray of exactly k elements.
💡 Key: Sliding window - remove left element, add new right element
Solution
def find_max_average(nums, k):
    window_sum = sum(nums[:k])
    max_sum = window_sum
    for i in range(k, len(nums)):
        window_sum += nums[i] - nums[i - k]
        max_sum = max(max_sum, window_sum)
    return max_sum / k

# Example: [1,12,-5,-6,50,3], k=4 → 12.75
⏱️ Time
O(n)
💾 Space
O(1)

35. Minimum Size Subarray Sum
Find minimum length subarray whose sum >= target value.
💡 Key: Sliding window - expand right, shrink left while sum >= target
Solution
def min_sub_array_len(target, nums):
    left = total = 0
    min_len = float('inf')
    for right in range(len(nums)):
        total += nums[right]
        while total >= target:
            min_len = min(min_len, right - left + 1)
            total -= nums[left]
            left += 1
    return 0 if min_len == float('inf') else min_len

# Example: target=7, [2,3,1,2,4,3] → 2
⏱️ Time
O(n)

36. Spiral Matrix II
Generate an n×n matrix filled with numbers 1 to n² in spiral order (clockwise).
💡 Key: Track boundaries (left, right, top, bottom) and shrink after each direction
Solution
def generate_matrix(n):
    res = [[0]*n for _ in range(n)]
    left, right, top, bottom = 0, n-1, 0, n-1
    num = 1
    
    while left <= right and top <= bottom:
        for j in range(left, right+1):
            res[top][j] = num
            num += 1
        top += 1
        for i in range(top, bottom+1):
            res[i][right] = num
            num += 1
        right -= 1
        if top <= bottom:
            for j in range(right, left-1, -1):
                res[bottom][j] = num
                num += 1
            bottom -= 1
        if left <= right:
            for i in range(bottom, top-1, -1):
                res[i][left] = num
                num += 1
            left += 1
    return res
⏱️ Time
O(n²)
💾 Space
O(n²)

37. Diagonal Traverse
Traverse all elements of m×n matrix diagonally, alternating up-right then down-left directions.
💡 Key: Group elements by (row+col) sum - same sum = same diagonal
Solution
def find_diagonal_order(mat):
    if not mat or not mat[0]:
        return []
    m, n = len(mat), len(mat[0])
    res = []
    
    for diag_sum in range(m + n - 1):
        diagonal = []
        for i in range(m):
            j = diag_sum - i
            if 0 <= j < n:
                diagonal.append(mat[i][j])
        if diag_sum % 2 == 1:
            diagonal.reverse()
        res.extend(diagonal)
    return res
⏱️ Time
O(m×n)
💾 Space
O(m×n)
Easy

def search_matrix(matrix, target):

💡 Key: Key property: mat[i][j] == mat[i-1][j-1]
Solution
def is_toeplitz_matrix(matrix):
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            if i > 0 and j > 0:
                if matrix[i][j] != matrix[i-1][j-1]:

    return True

41. Maximum Sum Circular Subarray
Find maximum sum of subarray in circular array (can wrap around from end to start).
💡 Key: Two Kadane passes: linear max and circular max (total - min subarray)
Solution
def max_subarray_sum_circular(nums):
    def kadane(arr, is_min=False):
        current = total = arr[0]
        for num in arr[1:]:
            current = min(num, current + num) if is_min else max(num, current + num)
            total = min(total, current) if is_min else max(total, current)
        return total
    
    linear_max = kadane(nums)
    circular_max = sum(nums) - kadane(nums, is_min=True)
    
    if circular_max == 0:
        return linear_max
    return max(linear_max, circular_max)
⏱️ Time
O(n)
💾 Space
O(1)
Medium
42. Rearrange Array Elements by Sign
Rearrange array to alternate positive and negative numbers. Assume equal count.
💡 Key: Separate into two lists, then merge alternately
Solution
def rearrange_by_sign(nums):
    pos = [x for x in nums if x > 0]
    neg = [x for x in nums if x < 0]
    res = []
    i = j = 0
    while i < len(pos) and j < len(neg):
        res.append(pos[i])
        res.append(neg[j])
        i += 1
        j += 1
    res.extend(pos[i:])
    res.extend(neg[j:])
    return res
⏱️ Time
O(n)
💾 Space
O(n)

46. Shortest Unsorted Continuous Subarray
Find shortest subarray that, if sorted, would make entire array sorted.
💡 Key: Find leftmost and rightmost breaks, then expand boundaries
Solution
def find_unsorted_subarray(nums):
    if len(nums) <= 1:
        return 0
    n = len(nums)
    left, right = 0, n - 1
    
    while left < n - 1 and nums[left] <= nums[left + 1]:
        left += 1
    if left == n - 1:
        return 0
    
    while right > 0 and nums[right] >= nums[right - 1]:
        right -= 1
    
    sub_min = min(nums[left:right + 1])
    sub_max = max(nums[left:right + 1])
    
    while left > 0 and nums[left - 1] > sub_min:
        left -= 1
    while right < n - 1 and nums[right + 1] < sub_max:
        right += 1
    
    return right - left + 1
⏱️ Time
O(n)
💾 Space
O(1)
Medium
47. Longest Mountain in Array
Find length of longest contiguous subarray forming mountain (strictly increase then strictly decrease).
💡 Key: For each peak, expand left (increase) and right (decrease)
Solution
def longest_mountain(arr):
    if len(arr) < 3:
        return 0
    max_len = 0
    for i in range(1, len(arr) - 1):
        if arr[i - 1] < arr[i] > arr[i + 1]:
            left = i - 1
            right = i + 1
            while left > 0 and arr[left - 1] < arr[left]:
                left -= 1
            while right < len(arr) - 1 and arr[right] > arr[right + 1]:
                right += 1
            max_len = max(max_len, right - left + 1)
    return max_len
⏱️ Time
O(n)
💾 Space
O(1)
Easy

49. Count Negative Numbers in Sorted Matrix
Count total negative numbers in m×n matrix where rows and columns are sorted in descending order.
💡 Key: Start top-right, move left if negative (all left are negative too)
Solution
def count_negatives(grid):
    count = 0
    row, col = 0, len(grid[0]) - 1
    while row < len(grid) and col >= 0:
        if grid[row][col] < 0:
            count += col + 1
            row += 1
        else:
            col -= 1
    return count
⏱️ Time

50. Count Good Triplets
Count triplets (i,j,k) where i
💡 Key: Brute force approach checking all triplets
Solution
def count_triplets(arr, distance):
    count = 0
    n = len(arr)
    for i in range(n):
        for j in range(i + 1, n):
            for k in range(j + 1, n):
                dist = abs(arr[i]-arr[j]) + abs(arr[j]-arr[k]) + abs(arr[i]-arr[k])
                if dist <= distance:
                    count += 1
    return count
⏱️ Time
O(n³)
💾 Space
O(1)
